import sqlite3
import customtkinter as ctk
from tkinter import messagebox, filedialog
import csv

DB_NAME = "shakha_management.db"

# ---------------------- DATABASE ----------------------

def create_table():
    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS swayamsevak (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT,
            age INTEGER,
            phone TEXT,
            address TEXT,
            education TEXT,
            occupation TEXT,
            blood_group TEXT,
            joining_date TEXT,
            shakha_name TEXT
        )
    """)
    conn.commit()
    conn.close()

# ---------------------- GUI FUNCTIONS ----------------------

def add_swayamsevak():
    if not entry_name.get():
        messagebox.showwarning("Validation", "Name is required.")
        return
    data = (
        entry_name.get(), entry_age.get(), entry_phone.get(), entry_address.get(),
        entry_education.get(), entry_occupation.get(), entry_blood.get(),
        entry_joining.get(), entry_shakha.get()
    )
    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    cursor.execute("""
        INSERT INTO swayamsevak
        (name, age, phone, address, education, occupation, blood_group, joining_date, shakha_name)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    """, data)
    conn.commit()
    conn.close()
    messagebox.showinfo("Success", "Swayamsevak added successfully!")
    clear_entries()
    load_data()

def load_data():
    for widget in frame_table.winfo_children():
        widget.destroy()

    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM swayamsevak")
    rows = cursor.fetchall()
    conn.close()

    for idx, row in enumerate(rows):
        ctk.CTkLabel(frame_table, text=str(row), anchor="w").grid(row=idx, column=0, sticky="w", padx=5)

def search_by_name():
    name = entry_search.get()
    if not name:
        messagebox.showwarning("Validation", "Enter name to search.")
        return
    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM swayamsevak WHERE name LIKE ?", ('%' + name + '%',))
    rows = cursor.fetchall()
    conn.close()

    for widget in frame_table.winfo_children():
        widget.destroy()

    if rows:
        for idx, row in enumerate(rows):
            ctk.CTkLabel(frame_table, text=str(row), anchor="w").grid(row=idx, column=0, sticky="w", padx=5)
    else:
        messagebox.showinfo("Result", "No record found.")

def delete_by_name():
    name = entry_search.get()
    if not name:
        messagebox.showwarning("Validation", "Enter name to delete.")
        return
    confirm = messagebox.askyesno("Confirm", f"Delete all records with name '{name}'?")
    if confirm:
        conn = sqlite3.connect(DB_NAME)
        cursor = conn.cursor()
        cursor.execute("DELETE FROM swayamsevak WHERE name = ?", (name,))
        conn.commit()
        conn.close()
        messagebox.showinfo("Deleted", f"Deleted records with name '{name}'.")
        load_data()

def upload_csv():
    path = filedialog.askopenfilename(filetypes=[("CSV files", "*.csv")])
    if not path:
        return
    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    with open(path, 'r') as file:
        reader = csv.reader(file)
        next(reader)
        for row in reader:
            if len(row) == 9:
                cursor.execute("""
                    INSERT INTO swayamsevak 
                    (name, age, phone, address, education, occupation, blood_group, joining_date, shakha_name)
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
                """, row)
    conn.commit()
    conn.close()
    messagebox.showinfo("Upload", "CSV data uploaded successfully.")
    load_data()

def clear_entries():
    entry_name.delete(0, 'end')
    entry_age.delete(0, 'end')
    entry_phone.delete(0, 'end')
    entry_address.delete(0, 'end')
    entry_education.delete(0, 'end')
    entry_occupation.delete(0, 'end')
    entry_blood.delete(0, 'end')
    entry_joining.delete(0, 'end')
    entry_shakha.delete(0, 'end')

# ---------------------- UI SETUP ----------------------

ctk.set_appearance_mode("dark")
ctk.set_default_color_theme("blue")

app = ctk.CTk()
app.title("Shakha Management System")
app.geometry("1000x700")

create_table()

frame_form = ctk.CTkFrame(app)
frame_form.pack(padx=20, pady=10, fill="x")

entry_name = ctk.CTkEntry(frame_form, placeholder_text="Name")
entry_age = ctk.CTkEntry(frame_form, placeholder_text="Age")
entry_phone = ctk.CTkEntry(frame_form, placeholder_text="Phone")
entry_address = ctk.CTkEntry(frame_form, placeholder_text="Address")
entry_education = ctk.CTkEntry(frame_form, placeholder_text="Education")
entry_occupation = ctk.CTkEntry(frame_form, placeholder_text="Occupation")
entry_blood = ctk.CTkEntry(frame_form, placeholder_text="Blood Group")
entry_joining = ctk.CTkEntry(frame_form, placeholder_text="Joining Date (YYYY-MM-DD)")
entry_shakha = ctk.CTkEntry(frame_form, placeholder_text="Shakha Name")

entries = [entry_name, entry_age, entry_phone, entry_address, entry_education,
           entry_occupation, entry_blood, entry_joining, entry_shakha]

for i, e in enumerate(entries):
    e.grid(row=i // 3, column=i % 3, padx=10, pady=5, sticky="ew")

btn_add = ctk.CTkButton(frame_form, text="Add Swayamsevak", command=add_swayamsevak)
btn_add.grid(row=3, column=2, pady=10, padx=10)

frame_action = ctk.CTkFrame(app)
frame_action.pack(padx=20, pady=5, fill="x")

entry_search = ctk.CTkEntry(frame_action, placeholder_text="Search/Delete by Name")
entry_search.grid(row=0, column=0, padx=5)

btn_search = ctk.CTkButton(frame_action, text="рџ”Ќ Search", command=search_by_name)
btn_delete = ctk.CTkButton(frame_action, text="рџ—‘пёЏ Delete", command=delete_by_name)
btn_csv = ctk.CTkButton(frame_action, text="рџ“‚ Upload CSV", command=upload_csv)
btn_refresh = ctk.CTkButton(frame_action, text="рџ”Ѓ Refresh", command=load_data)

btn_search.grid(row=0, column=1, padx=5)
btn_delete.grid(row=0, column=2, padx=5)
btn_csv.grid(row=0, column=3, padx=5)
btn_refresh.grid(row=0, column=4, padx=5)

frame_table = ctk.CTkScrollableFrame(app, height=400)
frame_table.pack(padx=20, pady=10, fill="both", expand=True)

load_data()
app.mainloop()
